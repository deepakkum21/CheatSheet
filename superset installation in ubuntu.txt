## Steps for superset installation in ubuntu

1. sudo apt install python3-pip
2. pip3 install virtualenv 
3. python3 -m venv supersetdata   or   apt-get install python3-venv
4. python3 -m venv supersetdata 
5. . supersetdata/bin/activate
6. pip install --upgrade setuptools pip
7. pip install apache-superset
8. superset db upgrade
9. export FLASK_APP=superset
10. superset fab create-admin
11. superset load_examples
12. superset init
13. superset run -p 8888 --with-threads --reload --debugger
